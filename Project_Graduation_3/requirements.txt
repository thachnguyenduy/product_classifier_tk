# ============================================
# Raspberry Pi 5 Installation Instructions
# ============================================
# 
# STEP 1: System dependencies
#   sudo apt-get update
#   sudo apt-get install -y python3-opencv python3-serial python3-pil python3-tk
#   sudo apt-get install -y python3-numpy
# 
# STEP 2: Install NCNN
#   pip3 install ncnn
#   
#   If pip install fails, build from source:
#   git clone https://github.com/Tencent/ncnn.git
#   cd ncnn
#   mkdir build && cd build
#   cmake -DCMAKE_BUILD_TYPE=Release -DNCNN_VULKAN=OFF -DNCNN_PYTHON=ON ..
#   make -j4
#   sudo make install
# 
# STEP 3: Install Python packages
#   pip3 install -r requirements.txt
# 
# STEP 4: Verify installation
#   python3 -c "import ncnn; print('NCNN OK')"
#   python3 -c "import cv2; print('OpenCV OK')"
#
